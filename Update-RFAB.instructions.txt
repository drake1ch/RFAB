@echo off
setlocal EnableExtensions DisableDelayedExpansion
chcp 1251 >nul

if "%REPO_URL%"=="" set "REPO_URL=https://github.com/drake1ch/RFAB.git"
if "%BRANCH%"=="" set "BRANCH=main"

if "%BASE_DIR%"=="" (
  echo [ОШИБКА] Не задан BASE_DIR.
  exit /b 1
)

if "%GIT_EXE%"=="" (
  set "GIT_EXE=%BASE_DIR%\PortableGit\cmd\git.exe"
)

set "PROFILE_SNAPSHOT_DIR=%TEMP%\rfab-updater-profiles-%RANDOM%-%RANDOM%"
set "GIT_DIR="
set "GIT_WORK_TREE="
set "REPO_KEY=%BASE_DIR%"
set "REPO_KEY=%REPO_KEY::=%"
set "REPO_KEY=%REPO_KEY:\=_%"
set "REPO_KEY=%REPO_KEY:/=_%"
set "REPO_KEY=%REPO_KEY: =_%"
set "REPO_KEY=%REPO_KEY:(=_%"
set "REPO_KEY=%REPO_KEY:)=_%"
set "REPO_KEY=%REPO_KEY:&=_%"
set "REPO_KEY=%REPO_KEY:;=_%"
set "FALLBACK_GIT_DIR=%LOCALAPPDATA%\RFAB\git\%REPO_KEY%"

if not exist "%GIT_EXE%" (
  echo [ОШИБКА] Portable Git не найден:
  echo   "%GIT_EXE%"
  exit /b 1
)

pushd "%BASE_DIR%" >nul || (
  echo [ОШИБКА] Не удалось перейти в папку:
  echo   "%BASE_DIR%"
  exit /b 1
)

echo [1/9] Сохраняем настройки пользователя...
call :snapshot_profile_lists || goto :fail

echo [2/9] Проверяем наличие RFAB...
call :ensure_git_metadata || goto :fail

echo [3/9] Настраиваем конфигурацию...
"%GIT_EXE%" config core.longpaths true >nul
"%GIT_EXE%" config credential.helper "" >nul 2>nul
"%GIT_EXE%" remote get-url origin >nul 2>nul
if errorlevel 1 (
  "%GIT_EXE%" remote add origin "%REPO_URL%" || goto :fail
) else (
  "%GIT_EXE%" remote set-url origin "%REPO_URL%" || goto :fail
)

echo [4/9] Скачиваем файлы...
"%GIT_EXE%" -c credential.helper= fetch --progress --prune origin "%BRANCH%" || goto :fail

echo [5/9] Подготавливаем установку...
"%GIT_EXE%" update-ref "refs/heads/%BRANCH%" "refs/remotes/origin/%BRANCH%" || goto :fail
"%GIT_EXE%" symbolic-ref HEAD "refs/heads/%BRANCH%" || goto :fail
"%GIT_EXE%" branch --set-upstream-to="origin/%BRANCH%" "%BRANCH%" >nul 2>nul

echo [6/9] Применяем изменения...
rem Do not touch PortableGit while this updater is running from it.
"%GIT_EXE%" restore --source="refs/heads/%BRANCH%" --staged --worktree -- ^
  . ^
  ":(exclude)PortableGit/**" ^
  ":(exclude)Update-RFAB.bat" ^
  ":(exclude,literal)MO2/ModOrganizer.exe" ^
  ":(exclude,literal)MO2/ModOrganizer.ini" ^
  ":(exclude,literal)MO2/logs/mo_interface.log" ^
  ":(exclude,literal)RFAB Launcher/RFAB Launcher Updater.exe" ^
  ":(exclude,literal)MO2/mods/RFAB/SKSE/Plugins/dtryKeyUtil/config/settings.ini" ^
  ":(exclude,literal)MO2/mods/[RFAB] Interface/MCM/Config/SkyUI_SE/settings.ini" ^
  ":(exclude,literal)MO2/mods/RFAB/SKSE/Plugins/StorageUtilData/RFAB_MCM_Settings.json" ^
  ":(exclude,literal)MO2/mods/RFAB/SKSE/Plugins/StorageUtilData/RFAB_XP_Settings.json" ^
  ":(exclude,literal)MO2/mods/RFAB/SKSE/Plugins/[RFAB] Dodge.ini" ^
  ":(exclude,literal)MO2/mods/RFAB/SKSE/Plugins/[RFAB] Extended Hotkeys.ini" ^
  ":(exclude,literal)MO2/mods/Unofficial Skyrim Special Edition Patch/Unofficial Skyrim Special Edition Patch.bsa" ^
  ":(exclude)MO2/profiles/**/SkyrimCustom.ini" ^
  ":(exclude)MO2/profiles/**/SkyrimPrefs.ini" ^
  ":(exclude,literal)MO2/mods/[RFAB] Interface/MCM/Settings/SkyUI_SE.ini" || goto :fail

rem This archive is distributed separately and should remain as a local ignored file.
"%GIT_EXE%" rm --cached --ignore-unmatch --quiet -- "MO2/mods/Unofficial Skyrim Special Edition Patch/Unofficial Skyrim Special Edition Patch.bsa" >nul 2>nul

echo [7/9] Настраиваем профиль MO2...
call :merge_profile_lists || goto :fail

echo Восстанавливаем настройки профиля...
call :restore_profile_inis || goto :fail

echo Проверяем профиль Default...
call :ensure_default_profile || goto :fail

echo [8/9] Копируем файлы...
call :seed_file "MO2\program-template\ModOrganizer.exe" "MO2\ModOrganizer.exe" "Mod Organizer" || goto :fail
call :seed_file "MO2\config-template\ModOrganizer.ini" "MO2\ModOrganizer.ini" "конфигурация Mod Organizer" || goto :fail
call :seed_file "MO2\settings-template\RFAB\settings.ini" "MO2\mods\RFAB\SKSE\Plugins\dtryKeyUtil\config\settings.ini" "настройки RFAB" || goto :fail
call :seed_file "MO2\settings-template\SkyUI_SE\settings.ini" "MO2\mods\[RFAB] Interface\MCM\Config\SkyUI_SE\settings.ini" "настройки SkyUI" || goto :fail
call :seed_file "MO2\settings-template\RFAB\RFAB_MCM_Settings.json" "MO2\mods\RFAB\SKSE\Plugins\StorageUtilData\RFAB_MCM_Settings.json" "настройки MCM RFAB" || goto :fail
call :seed_file "MO2\settings-template\RFAB\RFAB_XP_Settings.json" "MO2\mods\RFAB\SKSE\Plugins\StorageUtilData\RFAB_XP_Settings.json" "настройки опыта RFAB" || goto :fail
call :seed_file "MO2\settings-template\RFAB\[RFAB] Dodge.ini" "MO2\mods\RFAB\SKSE\Plugins\[RFAB] Dodge.ini" "настройки уклонения RFAB" || goto :fail
call :seed_file "MO2\settings-template\RFAB\[RFAB] Extended Hotkeys.ini" "MO2\mods\RFAB\SKSE\Plugins\[RFAB] Extended Hotkeys.ini" "горячие клавиши RFAB" || goto :fail
call :seed_file "MO2\settings-template\SkyUI_SE\SkyUI_SE.ini" "MO2\mods\[RFAB] Interface\MCM\Settings\SkyUI_SE.ini" "настройки MCM SkyUI" || goto :fail
call :seed_file "MO2\profile-template\Default\Skyrim.ini" "MO2\profiles\Default\Skyrim.ini" "настройки профиля Skyrim" || goto :fail
call :seed_file "MO2\profile-template\Default\SkyrimCustom.ini" "MO2\profiles\Default\SkyrimCustom.ini" "настройки профиля SkyrimCustom" || goto :fail
call :seed_file "MO2\profile-template\Default\SkyrimPrefs.ini" "MO2\profiles\Default\SkyrimPrefs.ini" "настройки профиля SkyrimPrefs" || goto :fail

echo [9/9] Готово.

set "EXIT_CODE=0"
goto :done

:ensure_git_metadata
if exist ".git\HEAD" (
  set "GIT_DIR="
  set "GIT_WORK_TREE="
  echo Метаданные репозитория найдены.
  goto :eof
)

if exist "%FALLBACK_GIT_DIR%\HEAD" (
  set "GIT_DIR=%FALLBACK_GIT_DIR%"
  set "GIT_WORK_TREE=%BASE_DIR%"
  echo Найдены пользовательские метаданные репозитория.
  goto :eof
)

echo Инициализируем репозиторий...
"%GIT_EXE%" init -b %BRANCH% >nul 2>nul
if not errorlevel 1 (
  set "GIT_DIR="
  set "GIT_WORK_TREE="
  echo Локальные git-метаданные инициализированы.
  goto :eof
)

echo Локальная папка .git недоступна для записи. Используем пользовательские метаданные.
if not exist "%FALLBACK_GIT_DIR%" mkdir "%FALLBACK_GIT_DIR%" || exit /b 1

set "GIT_DIR=%FALLBACK_GIT_DIR%"
set "GIT_WORK_TREE=%BASE_DIR%"

if not exist "%GIT_DIR%\HEAD" (
  "%GIT_EXE%" init -b %BRANCH% || exit /b 1
  echo Пользовательские git-метаданные инициализированы.
) else (
  echo Найдены пользовательские git-метаданные.
)
goto :eof

:snapshot_profile_lists
if not exist "MO2\profiles\*" (
  echo Существующие профили MO2 не найдены.
  exit /b 0
)

set "COPIED_ANY="
for /d %%P in ("MO2\profiles\*") do (
  if exist "%%~fP\modlist.txt" (
    if not exist "%PROFILE_SNAPSHOT_DIR%\%%~nxP" mkdir "%PROFILE_SNAPSHOT_DIR%\%%~nxP" || exit /b 1
    copy /Y "%%~fP\modlist.txt" "%PROFILE_SNAPSHOT_DIR%\%%~nxP\modlist.txt" >nul || exit /b 1
    set "COPIED_ANY=1"
  )
  if exist "%%~fP\loadorder.txt" (
    if not exist "%PROFILE_SNAPSHOT_DIR%\%%~nxP" mkdir "%PROFILE_SNAPSHOT_DIR%\%%~nxP" || exit /b 1
    copy /Y "%%~fP\loadorder.txt" "%PROFILE_SNAPSHOT_DIR%\%%~nxP\loadorder.txt" >nul || exit /b 1
    set "COPIED_ANY=1"
  )
  if exist "%%~fP\SkyrimPrefs.ini" (
    if not exist "%PROFILE_SNAPSHOT_DIR%\%%~nxP" mkdir "%PROFILE_SNAPSHOT_DIR%\%%~nxP" || exit /b 1
    copy /Y "%%~fP\SkyrimPrefs.ini" "%PROFILE_SNAPSHOT_DIR%\%%~nxP\SkyrimPrefs.ini" >nul || exit /b 1
    set "COPIED_ANY=1"
  )
  if exist "%%~fP\SkyrimCustom.ini" (
    if not exist "%PROFILE_SNAPSHOT_DIR%\%%~nxP" mkdir "%PROFILE_SNAPSHOT_DIR%\%%~nxP" || exit /b 1
    copy /Y "%%~fP\SkyrimCustom.ini" "%PROFILE_SNAPSHOT_DIR%\%%~nxP\SkyrimCustom.ini" >nul || exit /b 1
    set "COPIED_ANY=1"
  )
)

if defined COPIED_ANY (
  echo Списки профилей MO2 сохранены.
) else (
  echo Локальные modlist/loadorder не найдены.
)
exit /b 0

:merge_profile_lists
if not exist "%PROFILE_SNAPSHOT_DIR%\*" (
  echo Нет сохранённых списков MO2 для объединения.
  exit /b 0
)

powershell -NoProfile -ExecutionPolicy Bypass -Command ^
  "$ErrorActionPreference = 'Stop';" ^
  "$profiles = Join-Path $env:BASE_DIR 'MO2\profiles';" ^
  "$snapshot = $env:PROFILE_SNAPSHOT_DIR;" ^
  "$enc = New-Object System.Text.UTF8Encoding($false);" ^
  "function RL([string]$p){ if (Test-Path -LiteralPath $p) { [System.IO.File]::ReadAllLines($p) } else { @() } };" ^
  "function WL([string]$p, [string[]]$l){ $d = Split-Path -Path $p -Parent; if ($d -and -not (Test-Path -LiteralPath $d)) { New-Item -ItemType Directory -Path $d -Force | Out-Null }; [System.IO.File]::WriteAllLines($p, $l, $enc) };" ^
  "function MM([string]$c, [string]$u){ $cl = RL $c; $ul = RL $u; $mods = New-Object 'System.Collections.Generic.HashSet[string]' ([System.StringComparer]::OrdinalIgnoreCase); $out = New-Object 'System.Collections.Generic.List[string]'; foreach ($line in $cl) { $out.Add($line) | Out-Null; if ($line -match '^[\+\-]\s*(.+)$') { $n = $matches[1].Trim(); if ($n) { $mods.Add($n) | Out-Null } } }; foreach ($line in $ul) { if ($line -match '^[\+\-]\s*(.+)$') { $n = $matches[1].Trim(); if ($n -and -not $mods.Contains($n)) { $out.Add($line) | Out-Null; $mods.Add($n) | Out-Null } } }; WL $c $out.ToArray() };" ^
  "function ML([string]$c, [string]$u){ $cl = RL $c; $ul = RL $u; $seen = New-Object 'System.Collections.Generic.HashSet[string]' ([System.StringComparer]::OrdinalIgnoreCase); $out = New-Object 'System.Collections.Generic.List[string]'; foreach ($line in $cl) { $e = $line.Trim(); if ($e -and $seen.Add($e)) { $out.Add($e) | Out-Null } }; foreach ($line in $ul) { $e = $line.Trim(); if ($e -and $seen.Add($e)) { $out.Add($e) | Out-Null } }; WL $c $out.ToArray() };" ^
  "Get-ChildItem -Path $snapshot -Recurse -File | Where-Object { $_.Name -ieq 'modlist.txt' -or $_.Name -ieq 'loadorder.txt' } | ForEach-Object { $rel = $_.FullName.Substring($snapshot.Length).TrimStart('\'); $target = Join-Path $profiles $rel; $name = [System.IO.Path]::GetFileName($target); if (-not (Test-Path -LiteralPath $target)) { $dir = Split-Path -Path $target -Parent; if ($dir -and -not (Test-Path -LiteralPath $dir)) { New-Item -ItemType Directory -Path $dir -Force | Out-Null }; Copy-Item -LiteralPath $_.FullName -Destination $target -Force; return }; if ($name -ieq 'modlist.txt') { MM $target $_.FullName; return }; if ($name -ieq 'loadorder.txt') { ML $target $_.FullName } }" || exit /b 1

echo Списки MO2 успешно объединены.
exit /b 0

:restore_profile_inis
if not exist "%PROFILE_SNAPSHOT_DIR%\*" (
  echo Нет сохранённых ini-файлов профиля для восстановления.
  exit /b 0
)

for /d %%P in ("%PROFILE_SNAPSHOT_DIR%\*") do (
  if exist "%%~fP\SkyrimPrefs.ini" (
    if not exist "MO2\profiles\%%~nxP" mkdir "MO2\profiles\%%~nxP" || exit /b 1
    copy /Y "%%~fP\SkyrimPrefs.ini" "MO2\profiles\%%~nxP\SkyrimPrefs.ini" >nul || exit /b 1
  )
  if exist "%%~fP\SkyrimCustom.ini" (
    if not exist "MO2\profiles\%%~nxP" mkdir "MO2\profiles\%%~nxP" || exit /b 1
    copy /Y "%%~fP\SkyrimCustom.ini" "MO2\profiles\%%~nxP\SkyrimCustom.ini" >nul || exit /b 1
  )
)
echo Настройки профиля восстановлены.
exit /b 0

:ensure_default_profile
if exist "MO2\profiles\Default\*" (
  echo Профиль MO2 Default найден.
  exit /b 0
)

if not exist "MO2\profile-template\Default\*" (
  echo Шаблон профиля MO2 Default не найден. Пропускаем.
  exit /b 0
)

if not exist "MO2\profiles\Default" mkdir "MO2\profiles\Default" || exit /b 1
xcopy "MO2\profile-template\Default\*" "MO2\profiles\Default\" /E /I /Q /Y >nul || exit /b 1
echo Профиль MO2 Default создан из шаблона.
exit /b 0

:seed_file
set "SOURCE_FILE=%~1"
set "TARGET_FILE=%~2"
set "LABEL=%~3"

if exist "%TARGET_FILE%" (
  echo %LABEL% уже существует. Оставляем без изменений.
  exit /b 0
)

if not exist "%SOURCE_FILE%" (
  echo Шаблон для %LABEL% не найден. Пропускаем.
  exit /b 0
)

for %%I in ("%TARGET_FILE%") do set "TARGET_DIR=%%~dpI"
if not exist "%TARGET_DIR%" mkdir "%TARGET_DIR%" || exit /b 1

copy /Y "%SOURCE_FILE%" "%TARGET_FILE%" >nul || exit /b 1
echo %LABEL% создан.
exit /b 0

:fail
set "EXIT_CODE=%ERRORLEVEL%"

:done
popd >nul 2>nul
if exist "%PROFILE_SNAPSHOT_DIR%" rmdir /s /q "%PROFILE_SNAPSHOT_DIR%" >nul 2>nul
if not "%EXIT_CODE%"=="0" echo [ОШИБКА] Обновление завершилось с кодом %EXIT_CODE%.
exit /b %EXIT_CODE%
